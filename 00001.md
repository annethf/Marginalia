	void Even(int value)
	{
    	printf("请输入一个整数：\n");
    	scanf("%d",&value);

    	if (!(value % 2))
        	printf("%d 是偶数!\n",value);
	}

	int Max3(int a,int b,int c)
	{
    	printf("请输入三个整数(a,b,c):");
    	scanf("%d %d %d",&a,&b,&c);
    	return (((a > b) ? a : b) > c) ? ((a > b) ? a : b) : c;
	}

	void Sort4(int a,int b,int c,int d)

	void QuadraticSum(int a,int b)
	{
    	printf("请输入两个整数(a,b):");
    	scanf("%d%d",&a,&b);

    	if(a * a + b * b > 100)
        	printf("a * a + b * b=%d\n",a * a + b * b);
    	else
        	printf("a+b=%d\n",a+b);
	}

	void CubeDif(int a,int b)
	{
    	printf("请输入两个整数(a,b):");
    	scanf("%d%d",&a,&b);
	
    	if(a < b)
       		printf("a-b=%d\n",a-b);
    	else
        	printf("a^3-b^3=%d\n",a * a * a - b * b * b);
	}

	void Subtract(int a,int b)
	{
    	printf("请输入两个整数(a,b):");
    	scanf("%d%d",&a,&b);

    	if((a - b) % 2)
        	printf("a-b=%d\n",a-b);
    	else
        	printf("a-b的差不是奇数。\n");
	}
